import requests
from bs4 import BeautifulSoup
import pandas as pd
import json

# URL der Webseite
url = 'https://haushaltsdaten.odis-berlin.de/visualisierung?showExpenses=true&year=2022&modus=Funktionen'
page = requests.get(url)

soup = BeautifulSoup(page.content, 'html.parser')

# Den <script>-Tag mit den JSON-Daten finden
script_tag = soup.find("script", {"id": "__NEXT_DATA__"})
data = json.loads(script_tag.string)

# Extrahieren der relevanten Daten
categories_data = data['props']['pageProps']['hierarchyData']['children']

# Funktion zum rekursiven Summieren der Werte
def sum_values(item):
    total = 0
    if 'value' in item:
        total += item['value']
    if 'children' in item:
        for child in item['children']:
            total += sum_values(child)
    return total

# Die relevanten Hauptkategorien extrahieren
categories = []
for item in categories_data:
    category_name = item['name']
    category_value = sum_values(item)
    categories.append((category_name, category_value))

df = pd.DataFrame(categories, columns=["Kategorie", "Ausgaben"])

ausgaben_df['Ansatz 2020 in '] = ausgaben_df['Ansatz 2020 in '].str.replace('[.]', '', regex=True).str.replace(',', '.').astype(float)
ausgaben_df['Ansatz 2021 in '] = ausgaben_df['Ansatz 2021 in '].str.replace('[.]', '', regex=True).str.replace(',', '.').astype(float)

# Definieren der relevanten Hauptkategorien
hauptkategorien = [
    'Bildungswesen, Wissenschaft, Forschung, kulturelle Angelegenheiten',
    'Allgemeine Dienste',
    'Soziale Sicherung, Familie und Jugend, Arbeitsmarktpolitik',
    'Verkehrs- und Nachrichtenwesen',
    'Finanzwirtschaft',
    'Gesundheit, Umwelt, Sport und Erholung',
    'Wohnungswesen, Stdtebau, Raumordnung und kommunale Gemeinschaftsdienste',
    'Energie- und Wasserwirtschaft, Gewerbe, Dienstleistungen',
    'Ernhrung, Landwirtschaft und Forsten'
]

# Funktion zur Extraktion der Summe der Ausgaben fr jede Hauptkategorie
def extract_category_sums(df, year):
    categories = []
    for category in hauptkategorien:
        category_sum = df[df['Hauptfunktionsbezeichnung'] == category][f'Ansatz {year} in '].sum()
        categories.append((category, category_sum))
    return pd.DataFrame(categories, columns=["Kategorie", "Ausgaben"])

# Extrahieren der Summen fr die Hauptkategorien fr 2020 und 2021
df_2020_summary = extract_category_sums(ausgaben_df, 2020)
df_2021_summary = extract_category_sums(ausgaben_df, 2021)

import requests
from bs4 import BeautifulSoup
import pandas as pd
import json
import matplotlib.pyplot as plt

# Daten von 2022 abrufen
url_2022 = 'https://haushaltsdaten.odis-berlin.de/visualisierung?showExpenses=true&year=2022&modus=Funktionen'
page_2022 = requests.get(url_2022)
soup_2022 = BeautifulSoup(page_2022.content, 'html.parser')
script_tag_2022 = soup_2022.find("script", {"id": "__NEXT_DATA__"})
data_2022 = json.loads(script_tag_2022.string)
categories_data_2022 = data_2022['props']['pageProps']['hierarchyData']['children']

# Daten von 2023 abrufen
url_2023 = 'https://haushaltsdaten.odis-berlin.de/visualisierung?showExpenses=true&year=2023&modus=Funktionen'
page_2023 = requests.get(url_2023)
soup_2023 = BeautifulSoup(page_2023.content, 'html.parser')
script_tag_2023 = soup_2023.find("script", {"id": "__NEXT_DATA__"})

categories_data_2023 = data_2023['props']['pageProps']['hierarchyData']['children']

# Funktion zum rekursiven Summieren der Werte
def sum_values(item):
    total = 0
    if 'value' in item:
        total += item['value']
    if 'children' in item:
        for child in item['children']:
            total += sum_values(child)
    return total

# Extrahieren der relevanten Hauptkategorien
def extract_categories(categories_data):
    categories = []
    for item in categories_data:
        category_name = item['name']
        category_value = sum_values(item)
        categories.append((category_name, category_value))
    return pd.DataFrame(categories, columns=["Kategorie", "Ausgaben"])

df_2022_summary = extract_categories(categories_data_2022)
df_2023_summary = extract_categories(categories_data_2023)

# Krzen der Kategorienamen
def shorten_category_names(df):
    df['Kategorie'] = df['Kategorie'].str.replace('Bildungswesen, Wissenschaft, Forschung, kulturelle Angelegenheiten', 'Bildung')
    df['Kategorie'] = df['Kategorie'].str.replace('Allgemeine Dienste', 'Allg. Dienste')
    df['Kategorie'] = df['Kategorie'].str.replace('Soziale Sicherung, Familie und Jugend, Arbeitsmarktpolitik', 'S. Sicherung')
    df['Kategorie'] = df['Kategorie'].str.replace('Verkehrs- und Nachrichtenwesen', 'Verkehr')
    df['Kategorie'] = df['Kategorie'].str.replace('Finanzwirtschaft', 'Finanzen')
    df['Kategorie'] = df['Kategorie'].str.replace('Gesundheit, Umwelt, Sport und Erholung', 'Gesundheit')
    df['Kategorie'] = df['Kategorie'].str.replace('Wohnungswesen, Stdtebau, Raumordnung und kommunale Gemeinschaftsdienste', 'Wohnungswesen')
    df['Kategorie'] = df['Kategorie'].str.replace('Energie- und Wasserwirtschaft, Gewerbe, Dienstleistungen', 'Energie')
    df['Kategorie'] = df['Kategorie'].str.replace('Ernhrung, Landwirtschaft und Forsten', 'Ernhrung')
    return df

df_2022_summary = shorten_category_names(df_2022_summary)
df_2023_summary = shorten_category_names(df_2023_summary)

# Kombinieren der DataFrames
df_2022_summary['Jahr'] = 2022
df_2023_summary['Jahr'] = 2023
df_combined = pd.concat([df_2022_summary, df_2023_summary])

# Erstellen des Liniendiagramms
plt.figure(figsize=(14, 8))
for category in df_combined['Kategorie'].unique():
    data = df_combined[df_combined['Kategorie'] == category]
